def square(n):
	square=n*n
	print('The square of {0} is {1}'.format(n,square))
def cube(n):
	cube=n*n*n
	print('The cubeof {0} is {1}'.format(n,cube))

n=int(input('Enter a number:'))
square(n)
cube(n)
